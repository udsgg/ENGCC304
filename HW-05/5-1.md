```c++
#include <stdio.h>
#include <string.h> 

int main() {
    char empID[11];
    int Hrs;
    int salary;

    printf("Input the Employees ID(Max. 10 chars):\n");
    scanf("%s", empID);
    if (strlen(empID) > 11) { 
        printf("Error: Employee ID exceeds maximum length of 10 characters.\n"); 
        return 1;
    }
    printf("Input the working hrs:\n");
    scanf("%d", &Hrs);
    printf("Salary amount/hr:\n");
    scanf("%d", &salary);
    printf("\nEmployees ID = %s\n", empID);
    printf("Salary = %d\n", salary * Hrs);

    return 0;
}
```
