```c++
#include <stdio.h>

int main(){
    int score;
    printf("enter score: \n"); //ตรวจสอบว่าผู้ใช้ป้อนตัวเลขหรือไม่
    if(scanf("%d", &score ) != 1){
        printf("please enter number only.");
    }
    else if(score > 80){
        printf("A");
    }
    else if(score >= 75){
        printf("B+");
    }
    else if(score >= 70){
        printf("B");
    }
    else if(score >= 65){
        printf("c+");
    }
    else if(score >= 60){          
        printf("C");    
    }
    else if(score >= 55){
        printf("D+");
    }
    else if(score >= 50){
        printf("D");
    }
    else if(score < 50){
        printf("F");
    }
    else{
        printf("please enter number only.");
    }
}
```
