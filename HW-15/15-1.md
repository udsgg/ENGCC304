```c++
    #include <stdio.h>
    #include <ctype.h> // 

    int main() {
        char filename[100];
        FILE *file;
        int wordCount = 0;
        int inWord = 0;
        int ch;

        printf("Enter file name: \n");
        scanf("%s", filename);

        file = fopen(filename, "r"); // เปิดไฟล์ในโหมดอ่าน

        if (file == NULL) {  // ถ้าไม่มีไฟล์หรือเปิดไม่ได้
            printf("Error: ไม่สามารถเปิดไฟล์ '%s' ได้\n", filename);
            return 1;
        }

        while ((ch = fgetc(file)) != EOF) { // เริ่มอ่านไฟล์ทีละตัวอักษรจนกว่าจะหมด
            if (isalnum(ch)) { // ถ้าเจอตัวอักษรหรือเลข [ ถ้าอยากให้นับเฉพาะตัวอักษรให้ใช้ if(isalpha(ch)) ]
                if (inWord == 0) { // ถ้าสถานะการตรวจคำเป็น false
                    inWord = 1; // เจอกับสถานะการตรวจคำเป็น true
                    wordCount++; // เพิ่มค่าของการนับจำนวนคำ
                }
            } else { // ถ้าเจออย่างอื่นที่ไม่ใช่ตัวอักษร (สัญลักษณ์, ช่องว่าง)
                inWord = 0;
            }
        }

        fclose(file);
        printf("\nTotal number of words in '%s' : %d words\n", filename, wordCount);
        return 0;
    }
```
