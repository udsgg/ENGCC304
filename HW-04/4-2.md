```c++
#include <stdio.h>
#include <string.h> //เพิ่มไลบรารีสำหรับคำสั่ง strlen

int main() {
    char empID[11]; //กำหนดขนาด array ของ ID ให้เก็บได้ 10 ตัวอักษร + 1 ตัวสำหรับ null character
    int Hrs;
    int salary;

    printf("Input the Employees ID(Max. 10 chars):\n");
    scanf("%s", empID);
    if (strlen(empID) > 11) { //ฟังค์ชันตรวจสอบ char ของ ID ว่าเกิน 10 ตัวไหม
        printf("Error: Employee ID exceeds maximum length of 10 characters.\n"); 
        return 1;
    }
    printf("Input the working hrs:\n");
    scanf("%d", &Hrs);
    printf("Salary amount/hr:\n");
    scanf("%d", &salary);
    printf("\nEmployees ID = %s\n", empID);
    printf("Salary = %d\n", salary * Hrs);

    return 0;
}

```
