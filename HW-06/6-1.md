```c++
#include <stdio.h>

int main() {
    int num;
    printf("User Input : ");
    if (scanf("%d", &num) != 1) {
        printf("please enter number only.\n");
        return 0;
    }

    char *one[] = {"zero","one","two","three","four","five","six","seven","eight","nine"};
    char *teen[] = {"ten","eleven","twelve","thirteen","fourteen","fifteen","sixteen","seventeen","eighteen","nineteen"};
    char *ty[]   = {"","","twenty","thirty","forty","fifty","sixty","seventy","eighty","ninety"};

    printf("Output : ");

    if (num < 10) {
        for (int i = 0; i < 10; i++) {
            if (num == i) {
                printf("%s", one[i]);
                break;
            }
        }
    }
    else if (num < 20) {
        for (int i = 0; i < 10; i++) {
            if (num == 10 + i) {
                printf("%s", teen[i]);
                break;
            }
        }
    }
    else if (num < 100) {
        int t = num / 10;  // หลักสิบ
        int o = num % 10;  // หลักหน่วย

        for (int i = 2; i < 10; i++) {
            if (t == i) {
                printf("%s", ty[i]);
                break;
            }
        }

        if (o != 0) {
            for (int j = 1; j < 10; j++) {
                if (o == j) {
                    printf(" %s", one[j]);
                    break;
                }
            }
        }
    }
    else {
        printf("out of range");
    }

    printf("\n");
    return 0;
}
```
