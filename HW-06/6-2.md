```c++
#include <stdio.h>

int main(){
    printf("Enter value: \n");
    int value;
    scanf("%d", &value);
    printf("Series: ");
    if(value % 2 == 0){ //เช็คว่าค่าที่กรอกเป็นเลขคู่หรือไม่
        for(int i = value; i >= 0; i-=2){ //นับถอยหลังเลขคู่เริ่มตั้งแต่ค่ากรอก
                printf("%d ", i);
            }
        }
    else if(value % 2 != 0){ //เช็คว่าค่าที่กรอกเป็นเลขคี่หรือไม่
        for(int i = 1; i <= value; i+=2){ //นับเลขคี่ตั้งแต่ 1 ถึงค่าที่กรอก
            printf("%d ", i);
        }
    } 
}
```
