S students [3];

void grade(S inputgrade[]) {
    for (int s = 0; s < 5; s++) {
        if (inputgrade[s].ScoreSub1 >= 85 || inputgrade[s].ScoreSub2 >= 85 || inputgrade[s].ScoreSub3 >= 85 || inputgrade[s].ScoreSub4 >= 85 || inputgrade[s].ScoreSub5 >= 85) {
            sprintf(grades[s].Grade, "A+");
        }
        else if (inputgrade[s].ScoreSub1 >= 80 || inputgrade[s].ScoreSub2 >= 80 || inputgrade[s].ScoreSub3 >= 80 || inputgrade[s].ScoreSub4 >= 80 || inputgrade[s].ScoreSub5 >= 80) {
            sprintf(grades[s].Grade, "A");
        } 
        else if (inputgrade[s].ScoreSub1 >= 75 || inputgrade[s].ScoreSub2 >= 75 || inputgrade[s].ScoreSub3 >= 75 || inputgrade[s].ScoreSub4 >= 75 || inputgrade[s].ScoreSub5 >= 75) {
            sprintf(grades[s].Grade, "B+");
        }
        else if (inputgrade[s].ScoreSub1 >= 70 || inputgrade[s].ScoreSub2 >= 70 || inputgrade[s].ScoreSub3 >= 70 || inputgrade[s].ScoreSub4 >= 70 || inputgrade[s].ScoreSub5 >= 70) {
            sprintf(grades[s].Grade, "B");
        }  
        else if (inputgrade[s].ScoreSub1 >= 65 || inputgrade[s].ScoreSub2 >= 65 || inputgrade[s].ScoreSub3 >= 65 || inputgrade[s].ScoreSub4 >= 65 || inputgrade[s].ScoreSub5 >= 65) {
            sprintf(grades[s].Grade, "C+");
        }
        else if (inputgrade[s].ScoreSub1 >= 60 || inputgrade[s].ScoreSub2 >= 60 || inputgrade[s].ScoreSub3 >= 60 || inputgrade[s].ScoreSub4 >= 60 || inputgrade[s].ScoreSub5 >= 60) {
            sprintf(grades[s].Grade, "C");
        }  
        else if (inputgrade[s].ScoreSub1 >= 55 || inputgrade[s].ScoreSub2 >= 55 || inputgrade[s].ScoreSub3 >= 55 || inputgrade[s].ScoreSub4 >= 55 || inputgrade[s].ScoreSub5 >= 55) {
            sprintf(grades[s].Grade, "D+");
        } 
        else if (inputgrade[s].ScoreSub1 >= 50 || inputgrade[s].ScoreSub2 >= 50 || inputgrade[s].ScoreSub3 >= 50 || inputgrade[s].ScoreSub4 >= 50 || inputgrade[s].ScoreSub5 >= 50) {
            sprintf(grades[s].Grade, "D");
        }
        else {
            sprintf(grades[s].Grade, "F");
        }
    }
}
